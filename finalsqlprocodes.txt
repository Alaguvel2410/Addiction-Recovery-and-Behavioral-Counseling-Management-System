-- TRIGGER TO AUTO CALCULATE RESULTS
-- drop TRIGGER calculate_score;
DELIMITER //
CREATE TRIGGER calculate_score
AFTER INSERT ON responses
FOR EACH ROW
BEGIN
    INSERT INTO results(assessment_id,total_score,addiction_level,message)
    SELECT 
        r.assessment_id,
        SUM(o.score),
        CASE
            WHEN SUM(o.score) BETWEEN 20 AND 30 THEN 'No Addiction'
            WHEN SUM(o.score) BETWEEN 31 AND 45 THEN 'Low Addiction'
            WHEN SUM(o.score) BETWEEN 46 AND 60 THEN 'Moderate Addiction'
            WHEN SUM(o.score) BETWEEN 61 AND 80 THEN 'High Addiction'
        END,
        'This is only a self-check assessment,not the medical diagnosis'
    FROM responses r
    JOIN options_table o ON r.option_id = o.option_id
    WHERE r.assessment_id = NEW.assessment_id
    GROUP BY r.assessment_id
    ON DUPLICATE KEY UPDATE
        total_score = VALUES(total_score),
        addiction_level = VALUES(addiction_level),
        message = VALUES(message);
END //
DELIMITER ;

-- FINAL OUTPUT 
SELECT 
    u.user_id, u.name, u.type_of_addiction,
    CONCAT(r.total_score,' / 80') AS score,
    r.addiction_level,
    GROUP_CONCAT(t.tip_text SEPARATOR ' | ') AS tips
FROM results r
JOIN assessment a ON r.assessment_id = a.assessment_id
JOIN users u ON a.user_id = u.user_id
left JOIN tips t ON r.addiction_level = t.addiction_level
GROUP BY u.user_id,u.name,u.type_of_addiction,r.total_score,r.addiction_level;

FROM assessment;

SELECT MIN(total_score) AS minimum_score
FROM results;
-- Average score
SELECT AVG(total_score) AS average_score
FROM results;
-- Addiction level distribution
SELECT addiction_level, COUNT(*) AS number_of_users
FROM results
GROUP BY addiction_level;
